    夜幕行动 - 正式 - WEB

    sh /opt/www/scripts/game/ymxd/rsync_vue_game_admin.sh

    夜幕行动 - 正式 - API

    sh /opt/www/scripts/game/ymxd/rsync_xcp_api.sh

    sh /opt/www/scripts/game/ymxd/rsync_xcp_dcs.sh
